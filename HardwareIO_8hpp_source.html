60  message::platform::RawSensors::LEDPanel led_panel = {false, false, false};

68  struct Config {
69  Config() = default;

79  struct ServoState {
80  /// @brief True if new values should be written to the hardware.
81  bool dirty = false;
82 
83  /// @brief True if servo positions are simulated.
84  /// @details Note that commands can still be written to the hardware.
85  bool simulated = false;
86 
87  /// @brief Whether the servo torque is enabled.
88  bool torque_enabled = true;
89 
90  /// @brief Proportional gain. Cached value that is never read.
91  float p_gain = 32.0 / 255.0;
92  /// @brief Integral gain. Cached value that is never read.
93  float i_gain = 0.0f;
94  /// @brief Derivative gain. Cached value that is never read.
95  float d_gain = 0.0f;
96  /// @brief Speed of the servo. Cached value that is never read.
97  float moving_speed = 0.0f;
98  /// @brief Position the servo is attempting to reach. Cached value that is never read.
99  float goal_position = 0.0f;
100  /// @brief Torque of the servo. Cached value that is never read.
101  /// @details Ranges from 0.0 to 1.0.
102  float torque = 0.0f;
103 
104  /// @brief Current angle position of the servo. Simulated or read.
105  float present_position = 0.0f;
106  /// @brief Current speed of the servo. Simulated or read.
107  float present_speed = 0.0f;
108  /// @brief Current load of the servo. Simulated or read.
109  float load = 0.0f;
110  /// @brief Current voltage of the servo. Simulated or read.
111  float voltage = 10.0f;
112  /// @brief Current temperature of the servo. Simulated or read.
113  float temperature = 40.0f;
114  };
115 
116  /// @brief Stores the current state of the servos and the data being sent to the servos.
117  std::array<ServoState, 20> servo_state;
118 
119  public:
120  /// @brief called by a Powerplant to construct this reactor
121  explicit HardwareIO(std::unique_ptr<NUClear::Environment> environment);
122  };
