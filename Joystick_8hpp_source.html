19 /**
20  * Encapsulates all data relevant to a sampled joystick event.
21  */
22 class JoystickEvent {
23 private:
24  static constexpr unsigned char JS_EVENT_BUTTON{0x01}; // button pressed/released
25  static constexpr unsigned char JS_EVENT_AXIS{0x02}; // joystick moved
26  static constexpr unsigned char JS_EVENT_INIT{0x80}; // initial state of device
27 
28 public:
29  JoystickEvent() = default;
30 
31  /**
32  * The timestamp of the event, in milliseconds.
33  */
34  uint32_t time = 0;
35 
36  /**
37  * The value associated with this joystick event.
38  * For buttons this will be either 1 (down) or 0 (up).
39  * For axes, this will range between -32768 and 32767.
40  */
41  int16_t value = 0;
42 
43  /**
44  * The event type.
45  */
46  uint8_t type = 0;
47 
48  /**
49  * The axis/button number.
50  */
51  uint8_t number = 0;
52 
53  /**
54  * Returns true if this event is the result of a button press.
55  */
56  [[nodiscard]] bool isButton() const {
57  return (type & JS_EVENT_BUTTON) != 0;
58  }
59 
60  /**
61  * Returns true if this event is the result of an axis movement.
62  */
63  [[nodiscard]] bool isAxis() const {
64  return (type & JS_EVENT_AXIS) != 0;
65  }
66 
67  /**
68  * Returns true if this event is part of the initial state obtained when
69  * the joystick is first connected to.
70  */
71  [[nodiscard]] bool isInitialState() const {
72  return (type & JS_EVENT_INIT) != 0;
73  }
74 };
