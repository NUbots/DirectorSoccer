33  /**
34  * @brief Multinomial resampling. See [1] for details on the algorithm (a copy is in
35  * doc/Localisation/On_Resampling_Algorithms_for_Particle_Filters.pdf)
36  *
37  * [1]J. Hol, T. Schon and F. Dustafsson, "On Resampling Algorithms for Particle Filters", in 2006 IEEE Nonlinear
38  * Statistical Signal Processing Workshop, Cambridge. UK, 2006.
39  *
40  * @tparam Scalar The scalar type to used for the particle weights
41  */
42  template <typename Scalar>
43  struct Multinomial {
44  /**
45  * @brief Multinomial resampling weight generator
46  *
47  * @param count Number of particles being resampled
48  */
49  Multinomial(const int& /*count*/) : gen((std::random_device()())), dist(Scalar(0.0), Scalar(1.0)) {}
50 
51  /**
52  * @brief Calculate the next resampling weight
53  *
54  * @param i Particle index being resampling
55  */
56  [[nodiscard]] Scalar operator()(const int& /*i*/) {
57  return dist(gen);
58  }
59 
60  private:
61  /// @brief Uniform distribution on the interval [0, 1)
62  std::random_device rd;
63  std::mt19937 gen;
64  std::uniform_real_distribution<Scalar> dist;
65  };
66 
67  /**
68  * @brief Call the resampler algorithm using the above resampling weight calculation
69  *
70  * @param count Number of particles being resampled
71  * @param begin Iterator to the first weight to use for resampling
72  * @param end Iterator to the last weight to use for resampling
73  */
74  template <typename Iterator>
75  [[nodiscard]] std::vector<int> multinomial(const int& count, Iterator&& begin, Iterator&& end) {
76  return resample<Multinomial>(count, std::forward<Iterator>(begin), std::forward<Iterator>(end));
77  }
