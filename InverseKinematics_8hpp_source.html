46  /*! @brief Calculates the leg joints for a given input ankle position.
47  The robot coordinate system has origin a distance DISTANCE_FROM_BODY_TO_HIP_JOINT above the midpoint
48  of the hips. Robot coordinate system: x is out of the front of the robot y is left, from right shoulder
49  to left z is upward, from feet to head Input ankle coordinate system: x is forward, from heel to toe y is
50  left, z is normal to the plane of the foot
51  @param target The target 4x4 basis matrix for the ankle
52  @param isLeft Request for left leg motors or right leg motors?
53  @param RobotKinematicsModel The class containing the leg model of the robot.
54  */
55  template <typename Scalar>
56  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateLegJoints(
57  const message::actuation::KinematicsModel& model,
58  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& target_,
59  const LimbID& limb) {
60  const Scalar LENGTH_BETWEEN_LEGS = model.leg.LENGTH_BETWEEN_LEGS;
61  const Scalar DISTANCE_FROM_BODY_TO_HIP_JOINT = model.leg.HIP_OFFSET_Z;
62  const Scalar HIP_OFFSET_X = model.leg.HIP_OFFSET_X;
63  const Scalar UPPER_LEG_LENGTH = model.leg.UPPER_LEG_LENGTH;
64  const Scalar LOWER_LEG_LENGTH = model.leg.LOWER_LEG_LENGTH;
65 
66  std::vector<std::pair<ServoID, Scalar>> positions;
67 
68  Scalar hipYaw = 0;
69  Scalar hipRoll = 0;
70  Scalar hipPitch = 0;
71  Scalar knee = 0;
72  Scalar anklePitch = 0;
73  Scalar ankleRoll = 0;
74 
75  // Correct for input referencing the bottom of the foot
76  Eigen::Transform<Scalar, 3, Eigen::Isometry> target(target_);
77  target = target.translate(Eigen::Matrix<Scalar, 3, 1>(0.0, 0.0, model.leg.FOOT_HEIGHT));
78 
79  // Tci = transformation (not necessarily homogeneous) from input coordinates to calculation coordinates
80  // TODO remove this. It was due to wrong convention use
81  Eigen::Matrix<Scalar, 4, 4> Tci;
82  // clang-format off
83  Tci << 0.0, 1.0, 0.0, 0.0,
84  1.0, 0.0, 0.0, 0.0,
85  0.0, 0.0, -1.0, 0.0,
86  0.0, 0.0, 0.0, 1.0;
87  // clang-format on
88  // Rotate input position from standard robot coords to foot coords
89  Eigen::Matrix<Scalar, 3, 1> translation = (Tci * target.matrix().template rightCols<1>()).template head<3>();
90  target = Tci * target * Tci.transpose();
91  target.translation() = translation;
92 
93  // swap legs if needed
94  if (limb != LimbID::LEFT_LEG) {
95  target(0, 1) = -target(0, 1);
96  target(0, 2) = -target(0, 2);
97  target(1, 0) = -target(1, 0);
98  target(2, 0) = -target(2, 0);
99  target.translation().x() = -target.translation().x();
100  }
101 
102  const Eigen::Matrix<Scalar, 3, 1> ankleX = target.matrix().template leftCols<1>().template head<3>();
103  const Eigen::Matrix<Scalar, 3, 1> ankleY = target.matrix().template middleCols<1>(1).template head<3>();
104  const Eigen::Matrix<Scalar, 3, 1> anklePos = target.translation();
105 
106  const Eigen::Matrix<Scalar, 3, 1> hipOffset(LENGTH_BETWEEN_LEGS * 0.5,
107  HIP_OFFSET_X,
108  DISTANCE_FROM_BODY_TO_HIP_JOINT);
109 
110  Eigen::Matrix<Scalar, 3, 1> targetLeg = anklePos - hipOffset;
111 
112  const Scalar maxLegLength = UPPER_LEG_LENGTH + LOWER_LEG_LENGTH;
113  Scalar length = targetLeg.norm();
114  if (length > maxLegLength) {
115  targetLeg = targetLeg * maxLegLength / length;
116  length = targetLeg.norm();
117  }
118  const Scalar sqrLength = length * length;
119  const Scalar sqrUpperLeg = UPPER_LEG_LENGTH * UPPER_LEG_LENGTH;
120  const Scalar sqrLowerLeg = LOWER_LEG_LENGTH * LOWER_LEG_LENGTH;
121 
122  const Scalar cosKnee = (sqrUpperLeg + sqrLowerLeg - sqrLength) / (2.0 * UPPER_LEG_LENGTH * LOWER_LEG_LENGTH);
123  knee = std::acos(std::max(std::min(cosKnee, Scalar(1.0)), Scalar(-1.0)));
124 
125  const Scalar cosLowerLeg = (sqrLowerLeg + sqrLength - sqrUpperLeg) / (2.0 * LOWER_LEG_LENGTH * length);
126  const Scalar lowerLeg = std::acos(std::max(std::min(cosLowerLeg, Scalar(1.0)), Scalar(-1.0)));
127 
128  const Scalar phi2 = std::acos(targetLeg.dot(ankleY) / length);
129 
130  anklePitch = lowerLeg + phi2 - M_PI_2;
131 
132  const Eigen::Matrix<Scalar, 3, 1> unitTargetLeg = targetLeg / length;
133 
134  Eigen::Matrix<Scalar, 3, 1> hipX = ankleY.cross(unitTargetLeg);
135  Scalar hipXLength = hipX.norm();
136  if (hipXLength > 0) {
137  hipX /= hipXLength;
138  }
139  else {
140  NUClear::log<NUClear::DEBUG>(
141  "InverseKinematics::calculateLegJoints : targetLeg and ankleY parallel. This is unhandled at "
142  "the moment. requested pose = \n",
143  target.matrix());
144  return positions;
145  }
146  // Will be unit as ankleY and hipX are normal and unit
147  const Eigen::Matrix<Scalar, 3, 1> legPlaneTangent = ankleY.cross(hipX);
148 
149  ankleRoll = std::atan2(ankleX.dot(legPlaneTangent), ankleX.dot(hipX));
150 
151  const bool isAnkleAboveWaist = unitTargetLeg.dot(Eigen::Matrix<Scalar, 3, 1>::UnitZ()) < 0;
152 
153  const Scalar cosZandHipX = Eigen::Matrix<Scalar, 3, 1>::UnitZ().dot(hipX);
154  const bool hipRollPositive = cosZandHipX <= 0;
155  Eigen::Matrix<Scalar, 3, 1> legPlaneGlobalZ =
156  (isAnkleAboveWaist ? -1 : 1) * (Eigen::Matrix<Scalar, 3, 1>::UnitZ() - (cosZandHipX * hipX));
157  const Scalar legPlaneGlobalZLength = legPlaneGlobalZ.norm();
158  if (legPlaneGlobalZLength > 0) {
159  legPlaneGlobalZ /= legPlaneGlobalZLength;
160  }
161 
162  const Scalar cosHipRoll = legPlaneGlobalZ.dot(Eigen::Matrix<Scalar, 3, 1>::UnitZ());
163  hipRoll = (hipRollPositive ? 1 : -1) * std::acos(std::max(std::min(cosHipRoll, Scalar(1)), Scalar(-1)));
164 
165  const Scalar phi4 = M_PI - knee - lowerLeg;
166  // Superposition values:
167  const Scalar sinPIminusPhi2 = std::sin(M_PI - phi2);
168  const Eigen::Matrix<Scalar, 3, 1> unitUpperLeg =
169  unitTargetLeg * (std::sin(phi2 - phi4) / sinPIminusPhi2) + ankleY * (std::sin(phi4) / sinPIminusPhi2);
170  const bool isHipPitchPositive = hipX.dot(unitUpperLeg.cross(legPlaneGlobalZ)) >= 0;
171 
172  hipPitch = (isHipPitchPositive ? 1 : -1)
173  * std::acos(std::max(std::min(legPlaneGlobalZ.dot(unitUpperLeg), Scalar(1)), Scalar(-1)));
174 
175  // If leg is above waist then hipX is pointing in the wrong direction in the xy plane
176  Eigen::Matrix<Scalar, 3, 1> hipXProjected = (isAnkleAboveWaist ? -1 : 1) * hipX;
177  hipXProjected.z() = 0;
178  hipXProjected.normalize();
179  const bool isHipYawPositive = hipXProjected.dot(Eigen::Matrix<Scalar, 3, 1>::UnitY()) >= 0;
180  hipYaw = (isHipYawPositive ? 1 : -1)
181  * std::acos(std::max(std::min(hipXProjected.dot(Eigen::Matrix<Scalar, 3, 1>::UnitX()), Scalar(1)),
182  Scalar(-1)));
183 
184  if (limb == LimbID::LEFT_LEG) {
185  positions.push_back(std::make_pair(ServoID::L_HIP_YAW, -hipYaw));
186  positions.push_back(std::make_pair(ServoID::L_HIP_ROLL, hipRoll));
187  positions.push_back(std::make_pair(ServoID::L_HIP_PITCH, -hipPitch));
188  positions.push_back(std::make_pair(ServoID::L_KNEE, M_PI - knee));
189  positions.push_back(std::make_pair(ServoID::L_ANKLE_PITCH, -anklePitch));
190  positions.push_back(std::make_pair(ServoID::L_ANKLE_ROLL, ankleRoll));
191  }
192  else {
193  positions.push_back(std::make_pair(ServoID::R_HIP_YAW, (model.leg.LEFT_TO_RIGHT_HIP_YAW) * -hipYaw));
194  positions.push_back(std::make_pair(ServoID::R_HIP_ROLL, (model.leg.LEFT_TO_RIGHT_HIP_ROLL) * hipRoll));
195  positions.push_back(std::make_pair(ServoID::R_HIP_PITCH, (model.leg.LEFT_TO_RIGHT_HIP_PITCH) * -hipPitch));
196  positions.push_back(std::make_pair(ServoID::R_KNEE, (model.leg.LEFT_TO_RIGHT_KNEE) * (M_PI - knee)));
197  positions.push_back(
198  std::make_pair(ServoID::R_ANKLE_PITCH, (model.leg.LEFT_TO_RIGHT_ANKLE_PITCH) * -anklePitch));
199  positions.push_back(
200  std::make_pair(ServoID::R_ANKLE_ROLL, (model.leg.LEFT_TO_RIGHT_ANKLE_ROLL) * ankleRoll));
201  }
202 
203  return positions;
204  }

206  template <typename Scalar>
207  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateLegJoints(
208  const message::actuation::KinematicsModel& model,
209  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& leftTarget,
210  const Eigen::Transform<Scalar, 3, Eigen::Isometry>& rightTarget) {
211  auto joints = calculateLegJoints<Scalar>(model, leftTarget, LimbID::LEFT_LEG);
212  auto joints2 = calculateLegJoints<Scalar>(model, rightTarget, LimbID::RIGHT_LEG);
213  joints.insert(joints.end(), joints2.begin(), joints2.end());
214  return joints;
215  }

217  template <typename Scalar>
218  [[nodiscard]] std::vector<std::pair<ServoID, Scalar>> calculateHeadJoints(
219  const Eigen::Matrix<Scalar, 3, 1>& cameraUnitVector) {
220  return {std::make_pair(ServoID::HEAD_YAW, std::atan2(cameraUnitVector.y(), cameraUnitVector.x())),
221  std::make_pair(ServoID::HEAD_PITCH,
222  std::atan2(-cameraUnitVector.z(),
223  std::sqrt(cameraUnitVector.x() * cameraUnitVector.x()
224  + cameraUnitVector.y() * cameraUnitVector.y())))};
225  }
